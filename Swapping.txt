class Swaping{
    public static void main(String[] args) {
        int a = 20;
        int b =30;
        int c;

        System.out.println("Without swaping the value of elemets are "+a+" & "+b+"");
        a = b ;
        c=b;
        b=c;
        System.out.println("With swaping the value of elemets are "+a+" & "+b+"");

    }
}
